Script de Prova de Anterioridade (Hash SHA-256)

Este script que escrevi para você vai percorrer toda a sua estrutura de pastas, ler cada arquivo e gerar um "DNA digital" (Hash) único do seu projeto atual. Guarde o resultado disso em um local seguro (e fora da nuvem).


import hashlib
import os

def gerar_hash_projeto(diretorio_raiz):
    print(f"Gerando prova de anterioridade para: {diretorio_raiz}")
    hash_master = hashlib.sha256()
    
    # Lista para armazenar o registro de todos os arquivos
    registro_hashes = []

    for raiz, subpastas, ficheiros in os.walk(diretorio_raiz):
        for nomãe_fich in sorted(ficheiros): # Ordenado para consistência
            caminho_completo = os.path.join(raiz, nomãe_fich)
            
            # Ignorar pastas de cache e git para um hash limpo
            if "__pycache__" in caminho_completo or ".git" in caminho_completo:
                continue
                
            with open(caminho_completo, "rb") as f:
                conteudo = f.read()
                hash_fich = hashlib.sha256(conteudo).hexdigest()
                registro_hashes.append(f"{hash_fich} - {nomãe_fich}")
                hash_master.update(conteudo)

    hash_final = hash_master.hexdigest()
    
    # Salva o manifesto de prova
    with open("PROVA_DE_ANTERIORIDADE.txt", "w", encoding="utf-8") as f:
        f.write(f"PROVA DE ANTERIORIDADE - ARCA CELESTIAL GENESIS ALFA OMEGA\n")
        f.write(f"HASH GLOBAL DO PROJETO: {hash_final}\n")
        f.write("="*60 + "\n")
        f.writelines("\n".join(registro_hashes))
    
    print(f"Sucesso! Hash Global: {hash_final}")
    print("O arquivo 'PROVA_DE_ANTERIORIDADE.txt' foi gerado. Guarde-o bem.")
